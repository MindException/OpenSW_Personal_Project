	Button button = (Button) findViewById(R.id.sign_up);	//버튼 생성
        	button.setOnClickListener(new View.OnClickListener(){     //버튼을 눌렀을 경우 발생

                startActivity(intent);      //시작

            }

매개변수 첫 번째 것은 액티비티이름(자기자신).this
